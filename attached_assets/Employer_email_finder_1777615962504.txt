def prospeo_find_email(first_name, last_name, domain, api_key):
    if not api_key:
        return None
    try:
        # Based on current Prospeo API structure (name + domain)
        payload = {
            "first_name": first_name,
            "last_name": last_name,
            "domain": domain
        }
        resp = requests.post(
            "https://api.prospeo.io/v2/email-finder",  # adjust if endpoint changes
            headers={"Authorization": f"Bearer {api_key}", "Content-Type": "application/json"},
            json=payload,
            timeout=15
        )
        if resp.ok:
            data = resp.json()
            return data.get("email"), data.get("confidence", "N/A")
    except:
        pass
    return None, None

    if not prospeo_key or not domain: return None
    try:
        payload = {"first_name": first_name, "last_name": last_name, "domain": domain}
        r = requests.post("https://api.prospeo.io/email-finder",
                          headers={"Authorization": f"Bearer {prospeo_key}", "Content-Type": "application/json"},
                          json=payload, timeout=12)
        return r.json().get("email") if r.ok else None
    except:
        return None
